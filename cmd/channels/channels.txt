type hchan struct {
	qcount   uint           // колличество данных в буфере канала.
	dataqsiz uint           // размер буффера канала.
	buf      unsafe.Pointer // укозатель на буффер канала (последовательность байт в памяти).
	elemsize uint16         // Размер одного элемента типа *_type в байтах.
	closed   uint32         // флаг, который указывает закрыт ли канал или нет.
	elemtype *_type         // тип элементов в канале.
	sendx    uint           // индекс в буффере памяти (укозатель на конкретную ячейку) для записи след элемента.
	recvx    uint           // индекс в буффере памяти (укозатель на конкретную ячейку) для чтения след элемента.
	recvq    waitq  		// укозатель на односвязанный список горутин (очередь), которые ожидают чтение из канала.
	sendq    waitq  		// укозатель на односвязанный список горутин (очередь), которые ожидают записи в канал.
	lock     mutex          // блокировщик горутин (защита от конкурентного доступпа).
}
